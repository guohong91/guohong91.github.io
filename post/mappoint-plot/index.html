map+point plot

2021-04-16 1 min read

library(ggplot2)
library(scatterpie)
d<-read.table("clipboard",header=T)

#a white map
world <- map_data('world')
p <- ggplot(world, aes(long, lat)) +
geom_map(map=world, aes(map_id=region), fill="gray75", color="grey75") +
coord_quickmap()

#white map with point+Change the point size manually
p +geom_point(data=d,aes(x=long, y=lat,colour=factor(region),size=radius))+
scale_color_manual(d$region,values = c("#FF0000","#CCFF00","#00FF66","#0066FF","#CC00FF"))
