map+pie+point

2021-04-16 1 min read

library(ggplot2)
library(scatterpie)
d<-read.table("clipboard",header=T)

#a white map
world <- map_data('world')
p <- ggplot(world, aes(long, lat)) +
geom_map(map=world, aes(map_id=region), fill="gray75", color="grey75") +
coord_quickmap()

#white map with pie chart
p + geom_scatterpie(aes(x=long, y=lat, group=region, r=radius),
data=d, cols=c("IS","IG"), color=NA, alpha=.8) +
geom_scatterpie_legend(d$radius, x=-160, y=-55,n=3)

#white map with point
p + geom_point(aes(x=long, y=lat, color=region,size=radius),data=d)
